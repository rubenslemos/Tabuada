function toggleMostra(elementoAlvo) {
  elementoAlvo.addEventListener('click', () => {
    const elementosMostrados = [itensSoma, itensMenos, itensVezes, itensDividir, itensTodas];

    elementosMostrados.forEach((elemento) => {
      if (elemento === elementoAlvo) {
        elemento.classList.toggle("mostra");
      } else {
        elemento.classList.remove("mostra");
      }
    });
  });
}

toggleMostra(soma);
toggleMostra(menos);
toggleMostra(vezes);
toggleMostra(dividir);
toggleMostra(todas);
