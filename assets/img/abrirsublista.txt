abrirSoma = () => {
  soma.addEventListener('click', () => {
    itensSoma.classList.toggle("mostra")
    itensMenos.classList.remove("mostra")
    itensVezes.classList.remove("mostra")
    itensDividir.classList.remove("mostra")
    itensTodas.classList.remove("mostra")
  })
}
